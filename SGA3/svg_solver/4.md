Yes — I would frame it as a symmetry-reduction layer for a constraint-based SVG solver. In SVG, the <path> element is the generic shape primitive, and tools like SolveSpace explicitly export solved sketches to SVG, so the “solve first, serialize later” workflow is already natural in that ecosystem. 

Here is a standalone extension you can drop into the write-up:

import numpy as np
from scipy.optimize import least_squares

# A 6-fold symmetric radial profile:
# r(θ) = a + b cos(6θ)
# Think of (a, b) as the reduced coordinates of the shape.
# The symmetry group expands this one template into the full curve.

def radius(theta, a, b, n=6):
    return a + b * np.cos(n * theta)

def sample_points(a, b, n=6, m=360):
    theta = np.linspace(0.0, 2.0 * np.pi, m, endpoint=False)
    r = radius(theta, a, b, n=n)
    x = r * np.cos(theta)
    y = r * np.sin(theta)
    return x, y

def polygon_area(x, y):
    return 0.5 * abs(np.dot(x, np.roll(y, -1)) - np.dot(y, np.roll(x, -1)))

def perimeter(x, y):
    dx = np.diff(np.r_[x, x[0]])
    dy = np.diff(np.r_[y, y[0]])
    return np.sum(np.hypot(dx, dy))

def svg_path(x, y):
    parts = [f"M {x[0]:.2f},{y[0]:.2f}"]
    parts.extend(f"L {xi:.2f},{yi:.2f}" for xi, yi in zip(x[1:], y[1:]))
    parts.append("Z")
    return " ".join(parts)

TARGET_AREA = 12000.0
TARGET_PERIMETER = 500.0
TARGET_MIN_RADIUS = 30.0

def residuals(params):
    a, b = params
    x, y = sample_points(a, b, n=6)
    r = np.hypot(x, y)
    return np.array([
        polygon_area(x, y) - TARGET_AREA,
        perimeter(x, y) - TARGET_PERIMETER,
        np.min(r) - TARGET_MIN_RADIUS,
    ])

result = least_squares(
    residuals,
    x0=np.array([60.0, 10.0]),
    bounds=([1.0, 0.0], [500.0, 200.0]),
)

x, y = sample_points(*result.x, n=6)
print("Solved parameters:", result.x)
print("SVG path:", svg_path(x, y))

svg = f'''
<svg xmlns="http://www.w3.org/2000/svg" viewBox="-150 -150 300 300">
  <path d="{svg_path(x, y)}" fill="none" stroke="black" stroke-width="2"/>
</svg>
'''
print(svg)

The mechanism is simple: the solver works in reduced variables , and the symmetry group generates the full path from that small seed. In a fuller implementation, each orbit of the group action would share one parameter block, so the numerical optimizer only sees representatives, not every symmetric copy. That is the same reduction principle used by character decompositions: one mode in, many copies out.
